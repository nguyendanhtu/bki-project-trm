CREATE VIEW V_GD_GV_CONG_VIEC_MOI
AS

SELECT 
	cvm.ID_HOP_DONG_KHUNG,
	cvm.ID_NOI_DUNG_TT,
	cvm.SO_LUONG_HE_SO,
	cvm.DON_GIA,
	cvm.NGAY_DAT_HANG,
	cvm.NGAY_NGHIEM_THU,
	cvm.ID_TRANG_THAI,
	cvm.ID_USER_NHAP,
	cvm.GHI_CHU,
	hdk.SO_HOP_DONG,
	ndtt.TEN_NOI_DUNG,
	gv.HO_VA_TEN_DEM +' ' + gv.TEN_GIANG_VIEN 'HO_VA_TEN_GIANG_VIEN',
	nsd.TEN_TRUY_CAP,
	nsd.TEN
	

FROM GD_GV_CONG_VIEC_MOI cvm INNER JOIN DM_HOP_DONG_KHUNG hdk ON cvm.ID_HOP_DONG_KHUNG = hdk.ID
							 INNER JOIN DM_NOI_DUNG_THANH_TOAN ndtt ON cvm.ID_NOI_DUNG_TT = ndtt.ID
							 INNER JOIN DM_GIANG_VIEN gv ON gv.ID = hdk.ID_GIANG_VIEN
							 INNER JOIN HT_NGUOI_SU_DUNG nsd ON nsd.ID = cvm.ID_USER_NHAP
